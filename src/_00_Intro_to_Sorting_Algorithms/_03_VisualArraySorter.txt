int startIndex = 1;

void stepSort(int[] arr) {
  for (int i = startIndex; i < arr.length; i++) {
    if (arr[i - 1] > arr[i]) {
      int t = arr[i];
      arr[i] = arr[i - 1];
      arr[i - 1] = t;

      startIndex = i;
      return;
    }
  }
  startIndex = 1;
}
